CREATE OR REPLACE TABLE RETAIL_SALES (
    ORDER_ID STRING,
    ORDER_DATE DATE,
    MONTH_OF_SALE STRING,
    CUSTOMER_ID STRING,
    CUSTOMER_NAME STRING,
    COUNTRY STRING,
    REGION STRING,
    CITY STRING,
    CATEGORY STRING,
    SUBCATEGORY STRING,
    QUANTITY INT,
    DISCOUNT FLOAT,
    SALES FLOAT,
    PROFIT FLOAT
);
